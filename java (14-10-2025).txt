interface bank
{
    void banking();       
    void finance();
    void insurance();
}
abstract class goabanking implements bank{
    public void banking(){
        System.out.println("banking logics to be implemented");
    }
}
abstract class goafinance extends goabanking{
    public void finance(){
        System.out.println("finance logics to be implemented");
    }
}
class goainsurance extends goafinance{
    public void insurance(){
        System.out.println("insurance logics to be implemented");
    }
}
class example17 {
    public static void main(String[] args) {
        goainsurance cust1=new goainsurance();
        cust1.banking();
        cust1.finance();
        cust1.insurance();
    }
}
