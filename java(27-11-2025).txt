IQ: How do we add one collection to another?
-> There are 2 approaches to add 1 collection to another:
1) using addAll() method of collection.
2) Constructor based Approach


package collection.list;
import java.util.ArrayList;

public class ArrayList26 {
    public static void main(String[] args){
        ArrayList al=new ArrayList();
        al.add(100); al.add(200); al.add(300);
        System.out.println(al);
        //2 WAYS TO ADD COLLECTION IN ANOTHER COLLECTION----->1)addAll(al) approach   2)constructor based  approach
        ArrayList al2=new ArrayList();
        al2.add('b'); al2.add('m'); al2.add('w');
        al2.addAll(al);                          //      1st way
        System.out.println(al2);
        ArrayList al3=new ArrayList(al2);      //2nd way
        al3.add(true); al3.add(false);
        //      2nd way;
        System.out.println(al3);

    }
}

[b, m, w, 100, 200, 300]
[b, m, w, 100, 200, 300, true, false]
